FROM emp, salgrade
WHERE sal BETWEEN losal AND hisal
GROUP BY job;

4.
SELECT ename, sal, dname
FROM emp, dept
WHERE emp.deptno = dept.deptno AND (sal, dname) IN
	(SELECT MIN(sal), dname
	FROM emp, dept
	WHERE emp.deptno = dept.deptno
	GROUP BY dname);
